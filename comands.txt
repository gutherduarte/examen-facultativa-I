Comandos para crear base de datos:

entrar en la terminal del contenedor:

docker exec -it nombredecontenedor /bin/bash

dirigirse a la ruta:

cd /opt/mssql-tools/bin/

ejecutar herramienta pasandole variables de entorno:

./sqlcmd -S localhost -U sa -P Channel123*

crear base de datos:

CREATE DATABASE TodosDB;

GO

USE TodosDB;

GO

Insertar consulta que crea las tablas de la bd:

CREATE TABLE dbo.Todos (TodoID UNIQUEIDENTIFIER NOT NULL PRIMARY KEY, Author VARCHAR(150) NOT NULL, TodoDate DATETIME NOT NULL, TodoDescription VARCHAR(4000) NOT NULL, TodoState SMALLINT NOT NULL);

GO

salir de la herramienta y del contenedor con exit
